function doPost(e) {

  try {

    const data =
      JSON.parse(e.postData.contents);

    const sheet =
      SpreadsheetApp
      .openById("1ndgXDoLL4LoB3YWnSugfYINW5S8ouN8SlVLZsrkH7A8")
      .getSheetByName("Sheet1");

    const values =
      sheet.getDataRange().getValues();

    const headers = values[0];

    const articleCol =
      headers.indexOf("ARTICLE/ITEM") + 1;

    const remarksCol =
      headers.indexOf("REMARKS") + 1;

    const updatedByCol =
      headers.indexOf("UPDATED BY") + 1;

    const timestampCol =
      headers.indexOf("LAST UPDATE") + 1;

    const photoCol =
      headers.indexOf("PHOTOLINK") + 1;

    let targetRow = -1;

    for (let i = 1; i < values.length; i++) {

      if (
        String(values[i][articleCol - 1]).trim()
        ==
        String(data.article).trim()
      ) {
        targetRow = i + 1;
        break;
      }
    }

    let imageUrl = "";

    if (data.fileBytes && data.fileName) {

      const uploadUrl =
        "https://upload.imagekit.io/api/v1/files/upload";

      const auth =
        Utilities.base64Encode(
          "private_57I4************************:"
        );

      const payload = {
        file: data.fileBytes,
        fileName: data.fileName
      };

      const options = {
        method: "post",

        payload: payload,

        headers: {
          Authorization:
            "Basic " + auth
        },

        muteHttpExceptions: true
      };

      const response =
        UrlFetchApp.fetch(uploadUrl, options);

      const result =
        JSON.parse(response.getContentText());

      if (result.url) {
        imageUrl = result.url;
      }
    }

    sheet
      .getRange(targetRow, remarksCol)
      .setValue(data.remarks);

    sheet
      .getRange(targetRow, updatedByCol)
      .setValue(data.updatedby);

    sheet
      .getRange(targetRow, timestampCol)
      .setValue(data.timestamp);

    if (imageUrl) {

      sheet
        .getRange(targetRow, photoCol)
        .setValue(imageUrl);
    }

    return ContentService
      .createTextOutput(
        JSON.stringify({
          success: true,
          imageUrl: imageUrl
        })
      )
      .setMimeType(
        ContentService.MimeType.JSON
      );

  } catch(err) {

    return ContentService
      .createTextOutput(
        JSON.stringify({
          success: false,
          error: err.toString()
        })
      )
      .setMimeType(
        ContentService.MimeType.JSON
      );
  }
}
